const urlParams = new URLSearchParams(window.location.search); const classId = urlParams.get('class_id') || 'class_001'; document.addEventListener('DOMContentLoaded', async () => { const targetSelector = document.getElementById('target-selector'); const contentPanels = document.querySelectorAll('.content-panel'); const fileInput = document.getElementById('file-input'); const state = {}; // 教授用データを一括取得してバインドする本番処理 async function loadProfessorData() { const res = await fetch(`/api/classes/${classId}/professor-view`); const data = await res.json(); // メニューの動的構築 let menuHtml = `
  • 📊 クラス全体傾向・章管理
    • `; data.students.forEach(s => { menuHtml += `
  • 👨‍🎓 ${s.display_name}
    • `; }); targetSelector.innerHTML = menuHtml; // パネルのイベントリスナー再割り当て document.querySelectorAll('.target-item').forEach(item => { item.addEventListener('click', () => { document.querySelectorAll('.target-item').forEach(t => t.classList.remove('active')); contentPanels.forEach(p => p.classList.remove('active')); item.classList.add('active'); const targetId = item.getAttribute('data-target'); const pnl = document.getElementById(targetId); if(pnl) pnl.classList.add('active'); }); }); // 全体平均スコアの算出と描画 let sumK=0, sumT=0, sumA=0, count=0; data.students.forEach(s => { Object.values(s.enrollment.modules || {}).forEach(m => { sumK += m.current_status?.knowledge_level || 1; sumT += m.current_status?.thinking_level || 1; sumA += m.current_status?.application_level || 1; count++; }); }); const avgK = count > 0 ? sumK/count : 1; const avgT = count > 0 ? sumT/count : 1; const avgA = count > 0 ? sumA/count : 1; drawRadar("overall-radar", "overallRadar", state, [avgK, avgT, avgA], [4, 4, 3]); } // PDFアップロード本番処理 fileInput.addEventListener('change', async (e) => { if(e.target.files.length === 0) return; const formData = new FormData(); formData.append('file', e.target.files[0]); try { await fetch(`/api/classes/${classId}/modules/from-pdf`, { method: 'POST', body: formData }); alert("AIがシラバスモジュールを生成し、Firestoreに追加しました！"); location.reload(); } catch(err) { alert("解析に失敗しました。"); } }); await loadProfessorData(); }); // レーダー描画ロジック (app.jsと共通) function drawRadar(canvasId, stateKey, state, current, target) { const canvas = document.getElementById(canvasId); if (!canvas) return; if (state[stateKey]) { state[stateKey].data.datasets[0].data = current; state[stateKey].update(); return; } state[stateKey] = new Chart(canvas.getContext("2d"), { type: "radar", data: { labels: ["知識", "思考", "応用"], datasets: [ { label: "現在スコア", data: current, borderColor: "#f59e0b", backgroundColor: "rgba(245, 158, 11, 0.25)", borderWidth: 2 }, { label: "目標基準", data: target, borderColor: "#2a2d36", backgroundColor: "transparent", borderDash: [4, 4], borderWidth: 1.5, pointRadius: 0 } ] }, options: { responsive: true, maintainAspectRatio: false, scales: { r: { min: 0, max: 5, ticks: { display: false } } } } }); }
